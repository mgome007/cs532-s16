	function updateSearchString(action)
	{
		if (action == 'init' || action == 'blur') {
			if ($("form#search-form input[name=search_string]").val() == '') {
				$("form#search-form input[name=search_string]").addClass('input-info');
				$("form#search-form input[name=search_string]").val('optional');
			}
		} else if (action == 'focus') {
			if ($("form#search-form input[name=search_string]").hasClass('input-info')) {
				$("form#search-form input[name=search_string]").val('');
				$("form#search-form input[name=search_string]").removeClass('input-info');
			}
		}
	}

	function updateLocationString(action)
