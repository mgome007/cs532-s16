    function updateCounter(count) {
        var t = count.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',').split('');
        t = $.map(t, function(item) {
            return (item === ',' ? ',' : '<span>' + item + '</span>');
        }).join('');
        $('#downloadCounter').html(t + '+');
    }

    var total   = parseInt('100000', 10),
        count   = Math.floor(total / 1000 - 1) * 1000,
        counter = setInterval(function() {
            count++;
            if (count > total) {
                clearInterval(counter);
            } else {
                updateCounter(count);
            }
        }, 0.5);
    updateCounter(count);
});
