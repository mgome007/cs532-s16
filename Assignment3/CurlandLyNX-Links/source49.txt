     function toggle_submenu(link) {
        var menu_item = link.parentNode;
        var children = menu_item.getElementsByTagName('UL');
        var submenu = children[0];
        if (submenu.style.display) {
           submenu.style.display = '';
        } else {
           // If we're opening one up, close all the others
           var menu = document.getElementById('menu');
           var items = menu.getElementsByTagName('LI');
           for (var i = 0; i < items.length; i++) {
              var sub = items[i].getElementsByTagName('UL');
              if (sub[0]) {
                 sub[0].style.display = '';
              }
           }
           submenu.style.display = 'block';
        }
     }
