function PaintSlides(pObject,pCountry,pLanguage) {
  
      
      $.ajax({
        type: 'POST',
        url: '/ajax/sliders/',
        data: 'Country=' + pCountry + '&Language=' + pLanguage,
        beforeSend: function() {
          
          $(pObject).empty();
          
        },        
        success: function (response) {
          
          
          $(pObject).empty();
          
          if (response.success == "True") {
            
            //success
            $.each(response.recordset, function() { 
             
              
              var content = '';
              
              //if imagelink not null wrap in link
              if (this.imagelink.length > 0) {
                content = '<a href="' + this.imagelink + '" {0} >{1}</a>';  
              }
              
              //if this.newwindow = 0/1
              if (this.newwindow == "true") { 
                content = content.replace('{0}','target="_blank"');
              } else {
                content = content.replace('{0}','');              
              }
              
              
              //create image link
              var imagelink = '<img src="' + this.imageurl + '" />';                                
              
                
              //if link created, stuff image in it.
              if (content.length > 0) {
                content = content.replace('{1}',imagelink);
              } else {
                content = imagelink;
              }
              
              $(pObject).append('<li>' + content + '</li>');
              
              
            });
            
            
            jQuery('.flexslider').flexslider({
       
              start: function(slider){
                jQuery('body').removeClass('loading');
              }
            });
            
            
          } else {
            
            //general error message
          }
        },
        error: function () {  
         alert("error loading banners");
        }
      });
  
}
